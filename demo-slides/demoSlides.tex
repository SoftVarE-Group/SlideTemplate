\subsection{Colored Boxes}
\begin{frame}[fragile]{\insertsubsection\ with a really really really really really really really really really really really long title\hfill With Fill!}
	Normal versions:

\begin{mycolumns}[columns=3]
		\begin{definition}{A Definition}
			This is a definition.
\begin{minted}{java}
System.out.println("foo");
\end{minted}
		\end{definition}
\mynextcolumn
		\myexample{An Example}{This is an example.}
\mynextcolumn
		\begin{note}{A Note}
			This is a note.
		\end{note}
\end{mycolumns}
	\vfill

	Tight versions (e.g., for use with pictures):

	\leftmiddleandright{%
		\begin{definitiontight}{A Definition}
			This is a definition.
		\end{definitiontight}
	}{%
		\begin{exampletight}{An Example}
			This is an example.
		\end{exampletight}
	}{%
		\begin{notetight}{A Note}
			This is a note.
		\end{notetight}
	}
\end{frame}

\begin{frame}{Normal}
	\begin{mycolumns}[columns=5,widths={50},animation=keep]
		\mydefinition{X}{}
	\mynextcolumn
		\mydefinition{Y}{}
	\mynextcolumn
		\mydefinition{Z}{}
	\mynextcolumn
		\mydefinition{A}{}
	\mynextcolumn
		\mydefinition{B}{}
	\end{mycolumns}
	\begin{mycolumns}[columns=4,widths={30, 50},animation=forget,reverse]
		\mydefinition{X}{}
	\mynextcolumn
		\mydefinition{Y}{}
	\mynextcolumn
		\mydefinition{Z}{}
	\mynextcolumn
		\mydefinition{A}{}
	\end{mycolumns}
\end{frame}

\begin{frame}{No auto-width}
	\begin{mycolumns}[columns=3,widths={10,10,80},animation=keep,reverse]
		\mydefinition{X}{}
	\mynextcolumn
		\mydefinition{Y}{}
	\mynextcolumn
		\mydefinition{Z}{}
	\end{mycolumns}
	\begin{mycolumns}[columns=3,widths={10,10,80},animation=forget]
		\mydefinition{X}{}
	\mynextcolumn
		\mydefinition{Y}{}
	\mynextcolumn
		\mydefinition{Z}{}
	\end{mycolumns}
	Non-Reverse, does only show on last, which is probably not desired but part of the current implementation:
	\begin{mycolumns}[columns=3,widths={10,80,10},reverse]
		\mydefinition{X}{}
	\mynextcolumn
		\mydefinition{Y}{}
	\mynextcolumn
		\mydefinition{Z}{}
	\end{mycolumns}
\end{frame}
